${item.descripcion}
$${item.precio} c/u
${item.cantidad}
`; cont.appendChild(d); }); } function cambiarCantidad(index, delta) { let item = carrito[index]; let prod = inventarioLocal.find(x => x.codigo === item.codigo); let nuevaCant = item.cantidad + delta; if(nuevaCant > 0 && nuevaCant <= prod.stock) { item.cantidad = nuevaCant; renderizarCarrito(); } }
